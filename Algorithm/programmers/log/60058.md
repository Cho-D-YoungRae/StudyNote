###### My Solution 1
```python
def solution(p):
    if not p:
        return ''

    brackets_count = [0, 0]

    for i in range(len(p)):
        if p[i] == '(':
            brackets_count[0] += 1
        else:
            brackets_count[1] += 1

        if brackets_count[0] == brackets_count[1]:
            break

    u, v = p[:i+1], p[i+1:]

    if u[0] == '(':
        v = solution(v)
        return u + v
    else:
        result = "(" + solution(v) + ")"
        for i in range(1, len(u)-1):
            if u[i] == '(':
                result += ')'
            else:
                result += '('

        return result
```

def solution(p):
    if p=='': return p
    r=True; c=0
    for i in range(len(p)):
        if p[i]=='(': c-=1
        else: c+=1
        if c>0: r=False
        if c==0:
            if r:
                return p[:i+1]+solution(p[i+1:])
            else:
                return '('+solution(p[i+1:])+')'+''.join(list(map(lambda x:'(' if x==')' else ')',p[1:i]) ))
