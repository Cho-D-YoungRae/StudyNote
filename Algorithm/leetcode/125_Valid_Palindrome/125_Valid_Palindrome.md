#### [125_Valid_Palindrome](https://leetcode.com/problems/valid-palindrome/submissions/)


**첫 풀이**
```python
class Solution:
    def isPalindrome(self, s: str) -> bool:
        s = list(s.upper())
        result = []
        for ch in s:
            if ch.isalnum():
                result.append(ch)


        return result == result[::-1]
```

```python
import collections

class Solution:
    def isPalindrome(self, s: str) -> bool:
        dq = collections.deque()
        for ch in s:
            if ch.isalnum():
                dq.append(ch.lower())

        for _ in range(len(dq)//2):
            if dq.pop() != dq.popleft():
                return False

        return True
```
