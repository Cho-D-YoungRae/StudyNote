#### [14502. 연구소](https://www.acmicpc.net/problem/14502)

###### My solution
```python
from itertools import combinations
from collections import deque


if __name__ == '__main__':
    N, M = map(int, input().split())

    graph = []          # 입력
    empty_space = []    # 0 인 곳 좌표
    birus_start = []    # 2 인 곳 좌표
    for i in range(N):
        graph.append(list(map(int, input().split())))
        for j in range(M):
            if graph[i][j] == 0:
                empty_space.append((i, j))
            if graph[i][j] == 2:
                birus_start.append((i, j))

    di = [0, 1, 0, -1]
    dj = [-1, 0, 1, 0]

    # 바이러스가 너비우선으로 퍼진다
    # 퍼진 곳은 2로 체크
    def bfs(i, j):
        birusq = deque([(i, j)])
        while birusq:
            birus = birusq.popleft()
            for d in range(4):
                y, x = birus[0]+di[d], birus[1]+dj[d]
                if 0 <= y < N and 0 <= x < M and graph[y][x] == 0:
                    graph[y][x] = 2
                    birusq.append((y, x))



    max_safty = 0

    # 빈 공간 중 벽을 세울 3곳 
    for walls in combinations(empty_space, 3):
        # 벽은 1로
        for wall in walls:
            graph[wall[0]][wall[1]] = 1

        # 바이러스 시작점에서 확산 시작
        for birus in birus_start:
            bfs(birus[0], birus[1])

        # 안전한 공간 세기
        count_safty = 0

        # 안전한 곳이었던 곳 중에서
        for is_safty in empty_space:
            # 아직 안전하면 증가
            if graph[is_safty[0]][is_safty[1]] == 0:
                count_safty += 1
            # 0이 아니면 벽 혹은 바이러스.
            # 다음을 위해 다시 0으로 초기화
            else:
                graph[is_safty[0]][is_safty[1]] = 0

        max_safty = max(max_safty, count_safty)



    print(max_safty)

```
